import java.util.Map;
import java.util.HashMap;
import java.util.List;
import java.util.ArrayList;

class Solution {
    public int solution(int n, int[] lost, int[] reserve) {
        final Map<Integer, Integer> lostMap = new HashMap<>();
        for(int i : lost){
            lostMap.put(i, -1);
        }
        
        final Map<Integer, Integer> reserveMap = new HashMap<>();
        for(int i : reserve){
            if(lostMap.containsKey(i)){
                lostMap.remove(i);
                continue;
            }
            reserveMap.put(i, 1);
        }
        
        final Map<Integer, Integer> resultMap = new HashMap<>();
        for(int i = 0; i < n; i++){
            resultMap.put(i, lostMap.getOrDefault(i + 1, 0));
            if(reserveMap.containsKey(i + 1)){
                resultMap.put(i, reserveMap.getOrDefault(i + 1, 0));
            }
        }

        for(int i = 0; i < n; i++){
            if(resultMap.containsKey(i - 1)){
                if(resultMap.get(i) == -1){
                    if(resultMap.get(i - 1) == 1){
                        resultMap.put(i, 0);
                        resultMap.put(i - 1, 0);
                    }
                }
            }
            if(resultMap.containsKey(i + 1)){
                if(resultMap.get(i) == -1){
                    if(resultMap.get(i + 1) == 1){
                        resultMap.put(i, 0);
                        resultMap.put(i + 1, 0);
                    }
                }
            }
        }
        
        final List<Integer> list = new ArrayList(resultMap.values());
        
        int answer = 0;
        
        for(int i : list){
            if(i == 0 || i == 1){
                answer++;
            }
        }
        
        return answer;
    }
}
