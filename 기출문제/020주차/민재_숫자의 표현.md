class Solution {
    public int solution(int n) {
        int answer = 0;
        int low = 1;
        int sum = 0;
        for(int i = 1; i <= n; i++){
            sum += i;
            if(sum > n){
                while(sum > n){
                    sum -= low++;
                }
            }
            
            if(sum == n){
                answer++;
            }
        }
        return answer;
    }
}
