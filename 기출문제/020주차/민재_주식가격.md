import java.util.Stack;
class Solution {
    public int[] solution(int[] prices) {
        int max = prices.length;
        int[] answer = new int[max];
        Stack<Integer> stack = new Stack<>();

        for(int i = 0; i < max; i++){
            while(!stack.isEmpty() && prices[stack.peek()] > prices[i]){
                int idx = stack.pop();
                answer[idx] = i - idx;
            }
            stack.push(i);
        }

        while(!stack.isEmpty()){
            int idx = stack.pop();
            answer[idx] = max - 1 - idx;
        }
        return answer;
    }
}
