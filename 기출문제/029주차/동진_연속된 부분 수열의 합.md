class Solution {
    public int[] solution(int[] sequence, int k) {
        int[] answer = {};
        final int length = sequence.length;
        int sum = 0;
        int fir = 0;
        int sec = 0;
        int size = 0;
        
        int r1 = 0;
        int r2 = 0;
        
        for(int i = 0; i < length; i++) {
            sum += sequence[i];
            while(sum > k) {
                sum -= sequence[fir];
                fir++;
            }

            if(sum == k) {
                size = i - fir;
                if(length > size) {
                    r1 = fir;
                    r2 = i;
                } else if(length == size){
                    r1 = Math.min(r1, fir);
    				r2 = Math.min(r2, i);
                }
                
            }
        }
        return new int[]{r1, r2};
    }
}
