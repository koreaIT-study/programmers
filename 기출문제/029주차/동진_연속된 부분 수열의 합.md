class Solution {
    public int[] solution(int[] sequence, int k) {
        final int length = sequence.length;
        int sum = 0;
        int fir = 0;
        int sec = 0;
        int temp = sequence.length;
        
        int r1 = 0;
        int r2 = 0;
        
        for(int i = 0; i < length; i++) {
            sum += sequence[i];
            while(sum > k) {
                sum -= sequence[fir];
                fir++;
            }

            if(sum == k) {
                if(temp > i - fir) {
                    temp = i - fir;
                    r1 = fir;
                    r2 = i;
                } else if(temp == i - fir) {
                    r1 = Math.min(r1, fir);
    				r2 = Math.min(r2, i);
                }
            }
        }
        return new int[]{r1, r2};
    }
}
