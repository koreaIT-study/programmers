class Solution {
    public String solution(int n) {
        StringBuilder sb = new StringBuilder();
        String nums = "4124";
        int[] match = {3, 1, 2};
        long sum = 0;
        int len = 0;
        long pow = 0;
        int mod = 0;
        while(sum < n){
            sum += Math.pow(3, ++len);
        }

        for(int i = 0; i < len; i++){
            pow = (long)Math.pow(3, i+1);
            mod = (int)(n % pow);
            pow = (long)Math.pow(3, i);
            mod = (int)(mod / pow);
            n -= match[mod] * pow;
            sb.insert(0, nums.charAt(mod));
        }

        return sb.toString();
    }
}
