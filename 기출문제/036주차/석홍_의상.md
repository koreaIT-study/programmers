class Solution {
    fun solution(clothes: Array<Array<String>>): Int {
        val clothesMap = hashMapOf<String, Int>()
        clothes.forEach {
            var c = clothesMap.getOrDefault(it[1], 0)
            clothesMap[it[1]] = ++c
        }

        val result = clothesMap.values.fold(1) { total, num ->
            total * (num + 1)
        }

        return result - 1
    }
}
