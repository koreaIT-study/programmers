class Solution {
    public int solution(int n, long l, long r) {
        int answer = 0;
        for(long i = l; i <= r; i++){
            if(!dfs(i, n)){
                answer++;
            }
        }

        return answer;
    }

    private boolean dfs(long num, int n){
        long NUM = num - 1;
        int N = n - 1;
        if(n == 1){
            return NUM % 5 == 2;
        }

        long div = (long) Math.pow(5, N);
        long quotient = NUM / div;

        if(quotient == 2){
            return true;
        }

        return dfs(num - div * quotient, N);
    }
}

}
