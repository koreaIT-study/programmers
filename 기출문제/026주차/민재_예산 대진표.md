class Solution{
    public int solution(int n, int a, int b){
        int answer = 0;
        int div = 2;

        a -= 1;
        b -= 1;

        while((n /= 2) != 0){
            answer++;
            if(a / div == b / div){
                return answer;
            }
            div *= 2;
        }

        return answer;
    }
}
