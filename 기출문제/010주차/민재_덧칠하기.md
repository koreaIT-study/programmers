class Solution {
    public int solution(int n, int m, int[] section) {
        int answer = 0;
        boolean[] wall = new boolean[n+m];

        for(int part : section) wall[part] = true;

        for(int part : section) {
            if(!wall[part]) continue;
            for(int i = 0; i < m; i++){
                wall[part + i] = false;
            }
            answer++;
        }
        
        return answer;
    }
}
