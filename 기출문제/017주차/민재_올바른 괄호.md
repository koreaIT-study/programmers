import java.util.Stack;

class Solution {
    boolean solution(String s) {
        if (s.length() % 2 != 0) {
            return false;
        }

        char[] chars = s.toCharArray();
        Stack<Character> stack = new Stack<>();

        for (char c : chars) {
            if (c == '(') {
                stack.add(c);
            } else if(stack.isEmpty()){
                return false;
            }else{
                stack.pop();
            }
        }

        return stack.size() == 0;
    }
}
