class Solution {
    public int solution(int storey) {
        int answer = 0;

        while (storey != 0) {
            int num = storey % 10;
            answer += Math.min(10 - num, num);
            storey /= 10;
            if((num == 5 && goUp(storey)) || (num > (10 - num))){
                storey++;
            }
        }

        return answer;
    }

    private boolean goUp(int num){
        int temp = num % 10;
        if(temp < 5){
            return false;
        }

        return true;
    }
}
