### 나중에 다시 풀어봐야될듯
```java
class Solution {
    static int[][] sum;
    static int N, M;
 
    public static int solution(int[][] board, int[][] skill) {
        N = board.length;
        M = board[0].length;
        printArr(board);
        sum = new int[N + 1][M + 1];
        for (int[] s : skill) {
            int y1 = s[1], x1 = s[2];
            int y2 = s[3], x2 = s[4];
            int degree = s[5] * (s[0] == 1 ? -1 : 1);
            
            sum[y1][x1] += degree;
            sum[y1][x2 + 1] += (degree * -1);
            sum[y2 + 1][x1] += (degree * -1);
            sum[y2 + 1][x2 + 1] += degree;
        }
        
        System.out.println("=======");
        printArr(sum);
        System.out.println("=======");
        operate();
        printArr(sum);
        
        // 살아남은 건물 확인
        int answer = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                board[i][j] += sum[i][j];
                if (board[i][j] > 0) answer++;
            }
        }
        System.out.println("=======");
        printArr(board);
        return answer;
    }
    
    // 누적합 계산
    private static void operate() {
        // 상하
        for (int y = 1; y < N; y++) {
            for (int x = 0; x < M; x++) {
                sum[y][x] += sum[y - 1][x];
            }
        }
        printArr(sum);
        System.out.println("============");
        // 좌우
        for (int x = 1; x < M; x++) {
            for (int y = 0; y < N; y++) {
                sum[y][x] += sum[y][x - 1];
            }
        }
    }
    
    static public void printArr(int[][] arr){
        for(int[] ar : arr){
            System.out.println(Arrays.toString(ar));
        }
    }
}
```
